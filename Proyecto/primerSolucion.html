prefijoComunDos

function prefijoComunDos(str1,str2)

{

let result = "";

let n1 = str1.length, n2 = str2.length;

for (let i = 0, j = 0; i <= (n1 - 1) && j <= (n2 - 1); i++, j++) {

if (str1[i] != str2[j])

break;

result += str1[i];

}

return (result);

}
